  <script>

    // === Elite-Level State Management & Real-Time Precision ===

    class PayPerSecond {

      constructor() {

        this.state = {

          isRunning: false,

          rateType: 'hourly', // 'hourly' | 'daily'

          inputRate: 0,       // raw input (hourly or daily)

          perSecond: 0,       // computed $/sec

          accumulated: 0,     // total earned in session

          startTime: null,    // timestamp when clock started

          elapsedMs: 0,       // total elapsed on-clock milliseconds

          lastFrame: null,    // requestAnimationFrame timestamp

          sessionStart: null  // for session timer

        };


        this.elements = this.cacheDOM();

        this.bindEvents();

        this.loadPersistedState();

      }


      cacheDOM() {

        return {

          input: document.getElementById('rate-input'),

          rateBtns: document.querySelectorAll('.rate-btn'),

          toggle: document.getElementById('clock-toggle'),

          resetBtn: document.getElementById('reset-btn'),

          earnings: document.getElementById('earnings'),

          sessionTime: document.getElementById('session-time'),

          rateDisplay: document.getElementById('rate-display')

        };

      }


      bindEvents() {

        this.elements.input.addEventListener('input', () => this.updateRate());

        this.elements.rateBtns.forEach(btn => {

          btn.addEventListener('click', () => this.setRateType(btn.dataset.type));

        });

        this.elements.toggle.addEventListener('click', () => this.toggleClock());

        this.elements.resetBtn.addEventListener('click', () => this.resetCounter());

      }


      // === Persistence via localStorage (survives refresh) ===

      loadPersistedState() {

        const saved = localStorage.getItem('payPerSecond');

        if (saved) {

          const data = JSON.parse(saved);

          this.state.accumulated = data.accumulated || 0;

          this.state.elapsedMs = data.elapsedMs || 0;

          this.state.inputRate = data.inputRate || 0;

          this.state.rateType = data.rateType || 'hourly';

          this.elements.input.value = this.state.inputRate;

          this.updateRateDisplay();

          this.updateEarnings();

          this.updateSessionTime();

          this.highlightActiveRateBtn();

        }

      }


      saveState() {

        const toSave = {

          accumulated: this.state.accumulated,

          elapsedMs: this.state.elapsedMs,

          inputRate: this.state.inputRate,

          rateType: this.state.rateType

        };

        localStorage.setItem('payPerSecond', JSON.stringify(toSave));

      }


      // === Rate Logic ===

      updateRate() {

        const val = parseFloat(this.elements.input.value) || 0;

        this.state.inputRate = val;

        this.computePerSecond();

        this.updateRateDisplay();

        this.saveState();

      }


      setRateType(type) {

        this.state.rateType = type;

        this.computePerSecond();

        this.updateRateDisplay();

        this.highlightActiveRateBtn();

        this.saveState();

      }


      highlightActiveRateBtn() {

        this.elements.rateBtns.forEach(btn => {

          btn.classList.toggle('active', btn.dataset.type === this.state.rateType);

        });

      }


      computePerSecond() {

        if (this.state.rateType === 'hourly') {

          this.state.perSecond = this.state.inputRate / 3600;

        } else {

          this.state.perSecond = this.state.inputRate / (12 * 3600);

        }

      }


      updateRateDisplay() {

        const fmt = this.state.perSecond.toFixed(6).replace(/0+$/, '').replace(/\.$/, '');

        this.elements.rateDisplay.textContent = `$${fmt}/sec`;

      }


      // === Clock Control ===

      toggleClock() {

        if (this.state.isRunning) {

          this.stop();

        } else {

          this.start();

        }

      }


      start() {

        if (this.state.perSecond === 0) {

          alert('Please enter a valid rate first!');

          return;

        }


        this.state.isRunning = true;

        this.state.lastFrame = performance.now();

        this.state.startTime = this.state.lastFrame;

        this.state.sessionStart = this.state.sessionStart || Date.now();

        this.elements.toggle.textContent = 'Stop Earning';

        this.elements.toggle.classList.remove('off');

        this.elements.toggle.classList.add('on');

        this.animationFrame = requestAnimationFrame(this.tick);

      }


      stop() {

        this.state.isRunning = false;

        if (this.animationFrame) {

          cancelAnimationFrame(this.animationFrame);

        }

        this.elements.toggle.textContent = 'Start Earning';

        this.elements.toggle.classList.remove('on');

        this.elements.toggle.classList.add('off');

        this.saveState();

      }


      // === Reset Counter ===

      resetCounter() {

        if (this.state.isRunning) {

          if (!confirm('Clock is running. Reset will stop it and clear earnings. Continue?')) {

            return;

          }

          this.stop();

        }


        this.state.accumulated = 0;

        this.state.elapsedMs = 0;

        this.state.sessionStart = null;


        this.updateEarnings();

        this.updateSessionTime();

        this.saveState();

      }


      // === Real-Time Tick with requestAnimationFrame (60fps precision) ===

      tick = (now) => {

        if (!this.state.isRunning) return;


        const deltaMs = this.state.lastFrame ? (now - this.state.lastFrame) : 0;

        this.state.lastFrame = now;


        // Accumulate earnings

        const earnedThisFrame = this.state.perSecond * (deltaMs / 1000);

        this.state.accumulated += earnedThisFrame;

        this.state.elapsedMs += deltaMs;


        this.updateEarnings();

        this.updateSessionTime();

        this.saveState();


        this.animationFrame = requestAnimationFrame(this.tick);

      };


      // === UI Updates ===

      updateEarnings() {

        this.elements.earnings.textContent = `$${this.state.accumulated.toFixed(2)}`;

      }


      updateSessionTime() {

        const totalSec = Math.floor(this.state.elapsedMs / 1000);

        const h = String(Math.floor(totalSec / 3600)).padStart(2, '0');

        const m = String(Math.floor((totalSec % 3600) / 60)).padStart(2, '0');

        const s = String(totalSec % 60).padStart(2, '0');

        this.elements.sessionTime.textContent = `${h}:${m}:${s}`;

      }

    }


    // === Initialize App ===

    document.addEventListener('DOMContentLoaded', () => {

      window.app = new PayPerSecond();

    });

  </script>
